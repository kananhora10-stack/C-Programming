3. Write a C++ program that applies 10% discount using reference to update the original bill amount.

#include <iostream>
#include <iomanip>
using namespace std;
void applyDiscount(double &bill) {
    bill = bill * 90;
}
int main() {
    double billAmount;
    cout << "Enter the original bill amount: ";
    cin >> billAmount;
    cout << fixed << setprecision(2);
    cout << "Original bill: $" << billAmount << endl;
    applyDiscount(billAmount);
    cout << "Discounted bill (10% off): $" << billAmount << endl;
    return 0;
}
 
OUTPUT

5.write a c++ program to swap two numbers using pointers

#include <iostream>
using namespace std;
void swap(int *ptr1, int *ptr2) {
    int temp;
    temp = *ptr1;    
    *ptr1 = *ptr2;   
    *ptr2 = temp;    
}
int main() {
    int num1, num2;
    cout << "1st number: ";
    cin >> num1;
    cout << "2nd number: ";
    cin >> num2;
    cout << "\nBefore swapping:" << endl;
    cout << "num 1 = " << num1 << ", num 2 = " << num2 << endl;
    swap(&num1, &num2);
    cout << "\nAfter swapping:" << endl;
    cout << "num 1 = " << num1 << ", num 2 = " << num2 << endl;
    return 0;
}
 OUTPUT

After swapping:
num 1 = 7890, num 2 = 3456
