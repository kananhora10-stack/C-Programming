1.write a c++ program 5 stars in consisting pattern 4 rows and 5 columns:

#include <iostream>
using namespace std;

int main() {
    int rows = 4;
    int cols = 5;

    // Outer loop for rows
    for (int i = 0; i < rows; i++) {
        // Inner loop for columns
        for (int j = 0; j < cols; j++) {
            cout << "*";
        }
        // Move to the next line after each row
        cout << endl;
    }

    return 0;
}

4. MIRROR TRIANGLE =

#include <iostream>
using namespace std;

int main() {
    int rows = 5; // You can also take this as user input

    for(int i = 1; i <= rows; i++) {
        // Print leading spaces
        for(int j = 1; j <= rows - i; j++) {
            cout << " ";
        }
        // Print stars
        for(int k = 1; k <= i; k++) {
            cout << "*";
        }
        cout << endl;
    }
    return 0;
}

5.HOLLOW SQUARE PATTERN

#include <iostream>
using namespace std;

int main() {
    int size;
    cout << "Enter the side length of the square: ";
    cin >> size;

    // Outer loop for rows
    for (int i = 1; i <= size; i++) {
        // Inner loop for columns
        for (int j = 1; j <= size; j++) {
            // Check if current position is on the boundary
            if (i == 1 || i == size || j == 1 || j == size) {
                cout << "* ";
            } else {
                cout << "  "; // Two spaces to match the "* " width
            }
        }
        // Move to the next line after each row
        cout << endl;
    }

    return 0;
}

OUTPUT

* * * * * 
*       * 
*       * 
*       * 
* * * * *

6.PYRAMID PATTERN=

#include <iostream>
using namespace std;

int main() {
    int rows;
    cout << "Enter number of rows: ";
    cin >> rows;

    for(int i = 1; i <= rows; ++i) {
        // Print leading spaces
        for(int space = 1; space <= rows - i; ++space) {
            cout << "  ";
        }
        // Print stars
        for(int k = 0; k < 2 * i - 1; ++k) {
            cout << "* ";
        }
        cout << endl;
    }
    return 0;
}
